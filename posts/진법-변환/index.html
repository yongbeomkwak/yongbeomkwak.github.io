    #include <bits/stdc++.h>

    using namespace std;


    int calc(string s,const int b)
    {
        int ret=0;
        reverse(s.begin(),s.end());
        for(int i=0;i<s.size();i++)
        {
            int now=0;
            if('A'<=s[i]&&s[i]<='Z') now=10+s[i]-'A';
            else now=s[i]-'0';
            
            const int conv=(int)pow((double)b,(double)i) * now;
            ret+= conv;
        }
        
        return ret;
    }

    int main()
    {
        string n;
        int b;
        cin >> n >> b;
    
        cout << calc(n,b);

        return 0;
    }
